class Solution:
    def sumSubarrayMins(self, arr: List[int]) -> int:
        stack = []
        minSum = 0;

        for i in range(len(arr) + 1):
            while stack and (i == len(arr) or arr[stack[-1]] >= arr[i]):
                mid = stack.pop()
                l = -1 if not stack else stack[-1]
                r = i
                minSum += (((mid - l) * (r - mid)) * arr[mid])
            stack.append(i)
        
        return minSum % (10 ** 9 + 7)
