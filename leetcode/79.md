## 접근법

간단한 백트래킹 문제이다.

visited 관리 해주면서 쭉 백트래킹 해주면 된다.

## 코드

```py
class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        d = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        lenRow = len(board)
        lenCol = len(board[0])
        visited = [[False for _ in range(lenCol)] for _ in range(lenRow)]
        
        def dfs(row, col, now):
            if board[row][col] != word[now]:
                return False
            
            if now == len(word) - 1:
                return word[now] == board[row][col]
            
            result = False
            for dr, dc in d:
                nextRow = row + dr
                nextCol = col + dc
                if 0 <= nextRow < lenRow and 0 <= nextCol < lenCol:
                    if not visited[nextRow][nextCol]:
                        visited[nextRow][nextCol] = True
                        result |= dfs(nextRow, nextCol, now+1)
                        visited[nextRow][nextCol] = False

                        if result:
                            return True
            return result
        
        result = False
        for i in range(lenRow):
            for j in range(lenCol):
                visited[i][j] = True
                result |= dfs(i, j, 0)
                visited[i][j] = False
                
                if result:
                    return True
        return result
```
